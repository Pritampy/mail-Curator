  userId: string;
  name: string;

  action: FilterAction;
  category?: string;
  enabled: boolean;
  createdAt: Date;

}

  <name>Task 2: Create Rule model</name>

  <action>
Create Mongoose model for scheduled cleanup rules:

```typescript
// Schedule types
type ScheduleType = 'daily' | 'weekly' | 'monthly';

interface IRule extends Document {
  userId: string;
  name: string;
  conditions: IFilterCondition[]; // Reuse from Filter model
  action: 'delete' | 'archive' | 'skip' | 'label';
  label?: string;
  schedule: {
    type: ScheduleType;
    time: string; // HH:mm format
    dayOfWeek?: number; // 0-6 for weekly
    dayOfMonth?: number; // 1-31 for monthly
  };
  enabled: boolean;
  lastRun?: Date;
  nextRun?: Date;
  executionHistory: Array<{
    date: Date;
    emailsProcessed: number;
    success: boolean;
    error?: string;
  }>;
  createdAt: Date;
  updatedAt: Date;
}
```

Export Rule model, create indexes on {userId, enabled, nextRun}.
