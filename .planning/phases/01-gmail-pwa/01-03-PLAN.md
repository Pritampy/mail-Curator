<task type="auto">
  <name>Task 1: Create PWA web app manifest</name>
  <files>public/manifest.json</files>
  <action>
    Create public/manifest.json:
    1. name: "Kinetic - Gmail Inbox Cleaner"
    2. short_name: "Kinetic"
    3. description: "Clean your Gmail inbox with a swipe"
    4. start_url: "/"
    5. display: "standalone"
    6. background_color: "#1a1a1a" (use existing bg color)
    7. theme_color: "#d095ff" (use primary purple)
    8. icons: array with 192x192 and 512x512 placeholder paths
    9. scope: "/"
    10. orientation: "portrait"
  </action>
  <verify>
    <automated>grep -n "\"name\"" public/manifest.json</automated>
  </verify>
  <done>Valid manifest.json with installable PWA metadata</done>
</task>

<task type="auto">
  <name>Task 2: Create service worker for offline caching</name>
  <files>public/sw.js</files>
  <action>
    Create public/sw.js:
    1. Cache name: 'kinetic-v1'
    2. Install event: precache static assets (index.html, build files)
    3. Fetch event: network-first for API calls, cache-first for static assets
    4. Cache strategies:
       - API calls: network-first with cache fallback
       - Static files: cache-first with network update
    5. Skip waiting and claim clients on install
  </action>
  <verify>
    <automated>grep -n "fetch event" public/sw.js</automated>
  </verify>
  <done>Service worker caches app shell and handles API gracefully offline</done>
</task>

<task type="auto">
  <name>Task 3: Register service worker and add manifest link</name>
  <files>index.html</files>
  <read_first>index.html</read_first>
  <action>
    Update index.html to:
    1. Add <link rel="manifest" href="/manifest.json">
    2. Add <meta name="theme-color" content="#d095ff">
    3. Add <meta name="description" content="Clean your Gmail inbox with a swipe">
    4. Add service worker registration script in <head>:
       - if ('serviceWorker' in navigator) { navigator.serviceWorker.register('/sw.js') }
  </action>
  <verify>
    <automated>grep -n "serviceWorker" index.html</automated>
  </done>
  <done>Manifest linked and service worker registered</done>
</task>

<task type="auto">
  <name>Task 4: Configure Vite PWA plugin</name>
  <files>vite.config.ts</files>
  <read_first>vite.config.ts</read_first>
  <action>
    Update vite.config.ts:
    1. Install vite-plugin-pwa: npm install vite-plugin-pwa -D
    2. Add import: import { VitePWA } from 'vite-plugin-pwa'
    3. Add to plugins: VitePWA({
       registerType: 'autoUpdate',
       includeAssets: ['favicon.ico', 'robots.txt'],
       manifest: false (we use custom manifest.json)
     })
  </action>
  <verify>
    <automated>grep -n "VitePWA" vite.config.ts</automated>
  </done>
  <done>Vite configured for PWA with auto-update registration</done>
</task>

</tasks>
